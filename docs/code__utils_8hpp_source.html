79static HumanReadableTime human_readable_time(unsigned long long diff){
80 std::string unit;
81 std::string unit_fine;
82 unsigned long long diff_ns = diff;
83 unsigned long long diff_fine = diff;
84 if (diff/(1.e3l)<1.l) { unit = " ns"; }
85 else if (diff/1.e6l<1.l) {
86 unit = " µs";
87 diff /= 1.e3l;
88 }
89 else if (diff/1.e9l<1.l) {
90 unit = " ms";
91 diff /= 1.e6l;
92 unit_fine = " µs";
93 diff_fine /= 1.e3l;
94 }
95 else if (diff/(1.e9l*60.l)<1.l) {
96 unit = " s";
97 diff /= 1e9l;
98 unit_fine = " ms";
99 diff_fine /= 1e6l;
100 }
101 else if (diff/(1.e9l*3600.l)<1.l) {
102 unit = " m";
103 diff /= 6.e10l;
104 unit_fine = " s";
105 diff_fine /= 1.e9l;
106 }
107 else {
108 unit = " h";
109 diff /= 36.e11l;
110 unit_fine = " m";
111 diff_fine /= 6.e10l;
112 }
113 return {.unit=unit, .unit_fine=unit_fine, .diff=diff,
114 .diff_fine=diff_fine, .diff_ns=diff_ns};
